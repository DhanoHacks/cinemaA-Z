    /**
    * Gets the current state of the video. Returns value from the VideoState enum.
    * This is returned in the _vs key of the ping.
    * @return {string} The current video state. {@link VideoStrategy.VideoState}
    */
    VideoStrategy.prototype.getState = function() {
        //UNPLAYED: 's1',
        //PLAYED: 's2',
        //STOPPED: 's3',
        //COMPLETED: 's4'

        // uvp playback states
        //`-2` = EMPTY
        //`-1` = LOADING
        //`0` = STOPPED
        //`1` = PLAYING
        //`2` = PAUSED
        //`3` = BUFFERING
        let video_state = 's1'; // assume unplayed
        if (MetaC.Video.videoPlayer.getCurrentPlaybackState){
            let player_state = MetaC.Video.videoPlayer.getCurrentPlaybackState();
            if (player_state.playBackStarted != false){
                if (player_state.playBackComplete){
                    video_state = 's4';
                } else if (player_state.playbackState === 2){ // assuming PAUSED in uvp means STOPPED to cb
                    video_state = 's3';
                } else if (player_state.playbackState === 1){ // assuming PLAYING in uvp means PLAYED to cb
                    video_state = 's2';
                }
            }
        }

        return video_state;
    };
